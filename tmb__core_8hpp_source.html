683  /* ================== For parallel Hessian computation
684  Need three different parallel evaluation modes:
685  (1) *Parallel mode* where a parallel region is evaluated iff
686  current_parallel_region == selected_parallel_region
687  (2) *Serial mode* where all parallel region tests are evaluated
688  to TRUE so that "PARALLEL_REGION" tests are effectively removed.
689  A negative value of "current_parallel_region" or "selected_parallel_region"
690  is used to select this mode (the default).
691  (3) *Count region mode* where statements inside "PARALLEL_REGION{...}"
692  are *ignored* and "current_parallel_region" is increased by one each
693  time a parallel region is visited.
694  NOTE: The macro "PARALLEL_REGION" is supposed to be defined as
695  #define PARALLEL_REGION if(this->parallel_region())
696  where the function "parallel_region" does the book keeping.
697  */

700  int selected_parallel_region; /* Consider _this_ code-fragment */
701  int max_parallel_regions; /* Max number of parallel region identifiers,
702  e.g. max_parallel_regions=omp_get_max_threads();
703  probably best in most cases. */
704  bool parallel_region(){ /* Is this the selected parallel region ? */
705  bool ans;
706  if(current_parallel_region<0 || selected_parallel_region<0)return true; /* Serial mode */
707  ans = (selected_parallel_region==current_parallel_region) && (!parallel_ignore_statements);
708  current_parallel_region++;
709  if(max_parallel_regions>0)current_parallel_region=current_parallel_region % max_parallel_regions;
710  return ans;
711  }

715  selected_parallel_region=0;
716  parallel_ignore_statements=true; /* Do not evaluate stuff inside PARALLEL_REGION{...} */

1514 sphess MakeADHessObject2_(SEXP data, SEXP parameters, SEXP report, SEXP skip, int parallel_region=-1)
1515 {
1516  /* Some type checking */
1517  if(!Rf_isNewList(data))Rf_error("'data' must be a list");
1518  if(!Rf_isNewList(parameters))Rf_error("'parameters' must be a list");
1519  if(!Rf_isEnvironment(report))Rf_error("'report' must be an environment");

1522  objective_function< AD<AD<AD<double> > > > F(data,parameters,report);
1523  F.set_parallel_region(parallel_region);
1524  int n = F.theta.size();
1525  vector<bool> keepcol(n); // Scatter for fast lookup
1526  for(int i=0; i<n; i++){
1527  keepcol[i]=true;
1528  }
1529  for(int i=0; i<LENGTH(skip); i++){
1530  keepcol[INTEGER(skip)[i]-1]=false; // skip is R-index !
1531  }
1532 #define KEEP_COL(col) (keepcol[col])
1533 #define KEEP_ROW(row,col) ( KEEP_COL(row) & (row>=col) )
1534 
1535  /* Tape 1: Function R^n -> R */
1536  Independent(F.theta);
1537  vector< AD<AD<AD<double> > > > y(1);
1538  y[0] = F.evalUserTemplate();
1539  ADFun<AD<AD<double> > > tape1(F.theta, y);
1540 
1541  /* Tape 2: Gradient R^n -> R^n (and optimize) */
1542  vector<AD<AD<double> > > xx(n);
1543  for(int i=0; i<n; i++) xx[i] = CppAD::Value(F.theta[i]);
1544  vector<AD<AD<double> > > yy(n);
1545  Independent(xx);
1546  yy = tape1.Jacobian(xx);
1547  ADFun<AD<double > > tape2(xx,yy);
1548  if (config.optimize.instantly) tape2.optimize();
1549 
1550  /* Tape 3: Hessian R^n -> R^m (optimize later) */
1551  tape2.my_init(keepcol);
1552  int colisize;
1553  int m=0; // Count number of non-zeros (m)
1554  for(int i=0; i<int(tape2.colpattern.size()); i++){
1555  colisize = tape2.colpattern[i].size();
1556  if(KEEP_COL(i)){
1557  for(int j=0; j<colisize; j++){
1558  m += KEEP_ROW( tape2.colpattern[i][j] , i);
1559  }
1560  }
1561  }
1562  // Allocate index vectors of non-zero pairs
1563  vector<int> rowindex(m);
1564  vector<int> colindex(m);
1565  // Prepare reverse sweep for Hessian columns
1566  vector<AD<double> > u(n);
1567  vector<AD<double> > v(n);
1568  for(int i = 0; i < n; i++) v[i] = 0.0;
1569  vector<AD<double> > xxx(n);
1570  for(int i=0; i<n; i++) xxx[i]=CppAD::Value(CppAD::Value(F.theta[i]));
1571  vector<AD<double> > yyy(m);
1572  CppAD::vector<int>* icol;
1573  // Do sweeps and fill in non-zero index pairs
1574  Independent(xxx);
1575  tape2.Forward(0, xxx);
1576  int k=0;
1577  for(int i = 0; i < n; i++){
1578  if (KEEP_COL(i)) {
1579  tape2.myReverse(1, v, i /*range comp*/, u /*domain*/);
1580  icol = &tape2.colpattern[i];
1581  for(int j=0; j<int(icol->size()); j++){
1582  if(KEEP_ROW( icol->operator[](j), i )){
1583  rowindex[k] = icol->operator[](j);
1584  colindex[k] = i;
1585  yyy[k] = u[icol->operator[](j)];
1586  k++;
1587  }
1588  }
1589  }
1590  }
1591  ADFun< double >* ptape3 = new ADFun< double >;
1592  ptape3->Dependent(xxx,yyy);
1593  sphess ans(ptape3, rowindex, colindex);
1594  return ans;
1595 } // MakeADHessObject2
