4 /*
5  Given function f0. Define recursively higher order reverse
6  mode derivatives:
7 
8  f0: R^(n) -> R^(m) ( x -> f0 (x) )
9  f1: R^(n+m) -> R^(n) ( (x,w1) -> f0'(x)*w1 )
10  f2: R^(n+m+n) -> R^(n+m) ( (x,w1,w2) -> f1'(x)*w2 )
11  f3: R^(n+m+n+n+m) -> R^(n+m+n) ( (x,w1,w2,w3) -> f2'(x)*w3 )
12 
13  1. We define a 'generalized symbol' to represent all of these.
14  _Reverse_mode_AD_ is trivially obtained for this symbol by calling
15  itself on a higher level. Each occurance on the tape will occupy
16  O(n+m) memory units independent on the number of flops performed
17  by f0.
18 
19  2. _Double_versions_ of the generalized symbol are obtained using
20  nested AD types to tape f0, then recursively tape forward and
21  reverse mode sweeps.
22 
23  Finally, given (1) and (2) the macro TMB_ATOMIC_VECTOR_FUNCTION
24  will generate the atomic symbol.
25 */

77  template <class Base>
78  CppAD::ADFun<Base>* forrev(CppAD::ADFun<AD<Base> >* padf, vector<double> x_){
79  size_t n=padf->Domain();
80  size_t m=padf->Range();
81  vector<AD<Base> > x(n+m);
82  vector<AD<Base> > y(n);
83  for(int i=0;i<x_.size();i++)x[i]=AD<Base>(x_[i]);
84  for(int i=x_.size();i<x.size();i++)x[i]=AD<Base>(0);
85  vector<AD<Base> > tmp1(n);
86  vector<AD<Base> > tmp2(m);
87  CppAD::Independent(x);
88  for(size_t i=0;i<n;i++)tmp1[i]=x[i];
89  for(size_t i=0;i<m;i++)tmp2[i]=x[i+n];
90  padf->Forward(0,tmp1);
91  y = padf->Reverse(1,tmp2);
92  CppAD::ADFun<Base>* padf2=new CppAD::ADFun<Base>(x,y);
93  delete padf;
94  return padf2;
95  }

164  int get_output_dim(int input_dim){
165  int output_dim=-1;
166  // Fibonacci type recursion for each 'column'
167  if (input_dim == n) output_dim = m;
168  else if (input_dim == n+m) output_dim = n;
169  else if (input_dim == n+m+n) output_dim = n+m;
170  else if (input_dim == n+m+n+n+m) output_dim = n+m+n;
171  else Rf_error("get_output_dim failed");
172  return output_dim;
173  }
174  // Calculate level from input dimension
175  int get_level(int input_dim){
176  int level=-1;
177  if (input_dim == n) level = 0;
178  else if (input_dim == n+m) level = 1;
179  else if (input_dim == n+m+n) level = 2;
180  else if (input_dim == n+m+n+n+m) level = 3;
181  else Rf_error("get_level failed");
182  return level;
183  }

186  int level = get_level(tx.size());
