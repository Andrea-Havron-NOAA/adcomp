246 struct logIntegrate_t {
247  typedef Float Scalar;
248  global glob;
249  double mu, sigma, f_mu;
250  gk_config cfg;
251  double f(double x) {
252  Index k = glob.inv_index.size();
253  glob.value_inv(k - 1) = x;
254  glob.forward();
255  return glob.value_dep(0);
256  }
257  double g(double x) {
258  return (f(x + .5 * cfg.dx) - f(x - .5 * cfg.dx)) / cfg.dx;
259  }
260  double h(double x) {
261  return (g(x + .5 * cfg.dx) - g(x - .5 * cfg.dx)) / cfg.dx;
262  }
269  void rescale_integrand(const std::vector<ad_plain> &x) {
270  TMBAD_ASSERT(x.size() + 1 == glob.inv_index.size());
271  if (cfg.debug) Rcout << "rescale integrand:\n";
272  for (size_t i = 0; i < x.size(); i++) glob.value_inv(i) = x[i].Value();
273  mu = glob.value_inv(x.size());
274  f_mu = f(mu);
275  int i = 0;
276  for (; i < 100; i++) {
277  double g_mu = g(mu);
278  double h_mu = h(mu);
279  if (std::isfinite(f_mu) && !std::isfinite(h_mu)) {
280  cfg.dx = cfg.dx * .5;
281  continue;
282  }
283  double mu_new;
284  if (h_mu < 0)
285  mu_new = mu - g_mu / h_mu;
286  else
287  mu_new = mu + (g_mu > 0 ? cfg.dx : -cfg.dx);
288  double f_mu_new = f(mu_new);
289  if (cfg.debug) {
290  Rcout << "mu=" << mu << " mu_new=" << mu_new << " g_mu=" << g_mu
291  << " h_mu=" << h_mu << " f_mu=" << f_mu
292  << " f_mu_new=" << f_mu_new << "\n";
293  }
294  if (f_mu_new > f_mu + cfg.ytol) {
295  mu = mu_new;
296  f_mu = f_mu_new;
297  } else {
298  break;
299  }
300  }
301  sigma = 1. / sqrt(-h(mu));
302  if (!std::isfinite(sigma)) sigma = 10000;
303  if (cfg.debug)
304  Rcout << "==> i=" << i << " mu=" << mu << " f_mu=" << f_mu
305  << " sigma=" << sigma << "\n";
306  }

308  logIntegrate_t(global &glob, gk_config cfg)
309  : glob(glob), mu(0), sigma(1), f_mu(0), cfg(cfg) {
310  TMBAD_ASSERT(glob.inv_index.size() >= 1);
311  TMBAD_ASSERT(glob.dep_index.size() == 1);
312  }

316  Float operator()(Float u) {
317  Index k = glob.inv_index.size();
318  p_replay->value_inv(k - 1) = sigma * u + mu;
319  p_replay->forward(false, false);

321  if (cfg.nan2zero && ans != ans) ans = 0;
