47  template<class T>
48  T eval(T x_, T y_) {
49  double R = cfg.R;
50  int nrow = data.rows();
51  int ncol = data.cols();
52  T hx = (xmax - xmin) / (nrow - 1);
53  T hy = (ymax - ymin) / (ncol - 1);
54  // Transform (x_, y_) to 'lattice coordinates':
55  T i_ = (x_ - xmin) / hx;
56  T j_ = (y_ - ymin) / hy;
57  // Check
58  bool ok = (0 <= i_) && (i_ <= nrow-1) && (0 <= j_) && (j_ <= ncol-1);
59  if (!ok) return R_NaN;
60  // Envelope window: W := (i_, j_) + [-R, R]^2
61  // Get sub-lattice within envelope: (i_min:i_max) x (j_min:j_max)
62  int i_min = std::max(asDouble(i_) - R, (double) 0);
63  int j_min = std::max(asDouble(j_) - R, (double) 0);
64  int i_max = std::min(asDouble(i_) + R, (double) (nrow-1) );
65  int j_max = std::min(asDouble(j_) + R, (double) (ncol-1) );
66  // Loop through sub-lattice
67  T FWsum = 0, Wsum = 0;
68  for (int i = i_min; i <= i_max; i++) {
69  for (int j = j_min; j <= j_max; j++) {
70  T dist2 = sq( (T) i - i_ ) + sq( (T) j - j_ ) ;
71  double tiny = 1e-100;
72  T dist = sqrt( dist2 + tiny );
73  if (dist <= R) {
74  double F = data(i, j);
75  if (! ISNA(F) ) {
76  T W = kernel(dist / R);
77  FWsum += F * W;
78  Wsum += W;
79  }
80  }
81  }
82  }
83  return FWsum / Wsum;
84  }
