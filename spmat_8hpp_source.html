113 matrix<Type> discrLyap(matrix<Type> A_){
114  matrix<Type> I_(A_.rows(),A_.cols());
115  I_.setIdentity();
116  /* Sparse representations */
117  Eigen::SparseMatrix<Type> A=asSparseMatrix(A_);
118  Eigen::SparseMatrix<Type> I=asSparseMatrix(I_);
119  /* Kronecker */
120  Eigen::SparseMatrix<Type> AxA=kronecker(A,A);
121  Eigen::SparseMatrix<Type> IxI=kronecker(I,I);
122  matrix<Type> vecI=I_.vec().matrix();
123  Eigen::SparseMatrix<Type> C=IxI-AxA;
124  /* Solve system C*vecV=vecI.
125  Note: No assumptions on symmetry or eigenvalue-range of C. Therefore
126  rewrite it as (C'*C)*vecV=(C*vecI). Since (C'*C) is symmetric, the
127  LDL'-factorization can be applied.
128  */
129  Eigen::SimplicialLDLT< Eigen::SparseMatrix<Type> > ldlt(C.transpose()*C);
130  matrix<Type> z = ldlt.solve(C.transpose()*vecI);
131  matrix<Type> V(A_.rows(),A_.cols());
132  for(int i=0;i<z.size();i++)V(i)=z(i);
133  return V;
